Progressive enhancement: render every news item server-side, then collapse the overflow with a small inline script. With JavaScript the editor sees an "Initial visible" count and a "Load more" button that reveals N additional rows per click (settings.list.loadMoreCount). Without JavaScript every item stays visible and the button never appears, so search engines and screen readers see the complete list. The container exposes its config through data-attributes so editors can override per plugin instance.
Load more articles +{loadMoreCount}

Showing {initialCount} of {totalItems} articles

(function () { 'use strict'; if (window.__desiderioNewsLoadMore) { return; } window.__desiderioNewsLoadMore = true; function activate(root) { var grid = root.querySelector('[data-news-loadmore="grid"]'); var actions = root.querySelector('[data-news-loadmore="actions"]'); var trigger = root.querySelector('[data-news-loadmore="trigger"]'); var status = root.querySelector('[data-news-loadmore="status"]'); var counter = root.querySelector('[data-news-loadmore="counter"]'); if (!grid || !trigger || !actions) { return; } var initial = parseInt(root.getAttribute('data-initial-count'), 10) || 6; var step = parseInt(root.getAttribute('data-load-more-count'), 10) || 3; var total = parseInt(root.getAttribute('data-total-items'), 10) || 0; var items = grid.querySelectorAll('[data-news-loadmore="item"]'); if (items.length <= initial) { return; } var visible = initial; for (var i = initial; i < items.length; i += 1) { items[i].setAttribute('hidden', 'hidden'); } actions.removeAttribute('hidden'); if (counter) { counter.textContent = '+' + Math.min(step, items.length - visible); } updateStatus(); trigger.addEventListener('click', function () { var next = Math.min(visible + step, items.length); for (var j = visible; j < next; j += 1) { items[j].removeAttribute('hidden'); var firstFocusable = items[j].querySelector('a, button, [tabindex]:not([tabindex="-1"])'); if (j === visible && firstFocusable) { firstFocusable.setAttribute('data-news-loadmore-focus', 'true'); } } visible = next; updateStatus(); if (visible >= items.length) { trigger.setAttribute('disabled', 'disabled'); trigger.setAttribute('aria-disabled', 'true'); actions.setAttribute('hidden', 'hidden'); } else if (counter) { counter.textContent = '+' + Math.min(step, items.length - visible); } var focusTarget = grid.querySelector('[data-news-loadmore-focus="true"]'); if (focusTarget) { focusTarget.removeAttribute('data-news-loadmore-focus'); focusTarget.focus({ preventScroll: false }); } }); function updateStatus() { if (!status) { return; } status.textContent = status.getAttribute('data-template') ? status.getAttribute('data-template') .replace('{visible}', String(Math.min(visible, total))) .replace('{total}', String(total)) : 'Showing ' + Math.min(visible, total) + ' of ' + total + ' articles'; } } function init() { var roots = document.querySelectorAll('[data-news-loadmore="root"]'); Array.prototype.forEach.call(roots, activate); } if (document.readyState === 'loading') { document.addEventListener('DOMContentLoaded', init); } else { init(); } })();
